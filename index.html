function runProgram(start) { cpu.pc = start; cpu.cy = 0; cpu.z = 0; cpu.s = 0; cpu.p = 0; cpu.ac = 0; // AC added for DAA let limit = 0; // --- Helper: Register & Memory Access --- const getReg = (code) => { if (code === 0) return cpu.b; if (code === 1) return cpu.c; if (code === 2) return cpu.d; if (code === 3) return cpu.e; if (code === 4) return cpu.h; if (code === 5) return cpu.l; if (code === 6) return ram[(cpu.h << 8) | cpu.l] || 0; if (code === 7) return cpu.a; }; const setReg = (code, val) => { val &= 0xFF; if (code === 0) cpu.b = val; else if (code === 1) cpu.c = val; else if (code === 2) cpu.d = val; else if (code === 3) cpu.e = val; else if (code === 4) cpu.h = val; else if (code === 5) cpu.l = val; else if (code === 6) ram[(cpu.h << 8) | cpu.l] = val; else if (code === 7) cpu.a = val; }; // --- Helper: 16-Bit Pair Access --- const getPair = (p) => { if(p === 0) return (cpu.b << 8) | cpu.c; if(p === 1) return (cpu.d << 8) | cpu.e; if(p === 2) return (cpu.h << 8) | cpu.l; if(p === 3) return cpu.sp; }; const setPair = (p, val) => { val &= 0xFFFF; if(p === 0) { cpu.b = val >> 8; cpu.c = val & 0xFF; } else if(p === 1) { cpu.d = val >> 8; cpu.e = val & 0xFF; } else if(p === 2) { cpu.h = val >> 8; cpu.l = val & 0xFF; } else if(p === 3) { cpu.sp = val; } }; // --- Helper: Flag Updates --- // Note: In 8085, Z, S, P are updated together. CY is handled per-operation. const updateFlags = (val) => { cpu.z = (val === 0) ? 1 : 0; cpu.s = (val & 0x80) ? 1 : 0; let p = 0; for(let i=0; i<8; i++) { if((val>>i)&1) p++; } cpu.p = (p % 2 === 0) ? 1 : 0; }; const push = (val1, val2) => { ram[--cpu.sp] = val1; ram[--cpu.sp] = val2; }; const pop = () => { let l = ram[cpu.sp++]; let h = ram[cpu.sp++]; return {h, l}; }; while (limit < 100000) { // Increased safety limit for nested loops like Sorting let currOp = ram[cpu.pc++]; if (currOp === 0x76) { break; // HLT } else if (currOp === 0x00) { continue; // NOP } // 8-Bit Data Transfer (MOV) else if (currOp >= 0x40 && currOp <= 0x7F && currOp !== 0x76) { let dest = (currOp >> 3) & 0x07; let src = currOp & 0x07; setReg(dest, getReg(src)); } // LXI - Load Register Pair Immediate else if ((currOp & 0xCF) === 0x01) { let p = (currOp >> 4) & 0x03; let l = ram[cpu.pc++]; let h = ram[cpu.pc++]; setPair(p, (h << 8) | l); } // INX - Increment Register Pair else if ((currOp & 0xCF) === 0x03) { let p = (currOp >> 4) & 0x03; setPair(p, getPair(p) + 1); } // DCX - Decrement Register Pair else if ((currOp & 0xCF) === 0x0B) { let p = (currOp >> 4) & 0x03; setPair(p, getPair(p) - 1); } // DAD - Double Add (16-bit Addition) else if ((currOp & 0xCF) === 0x09) { let p = (currOp >> 4) & 0x03; let res = getPair(2) + getPair(p); // HL + Pair cpu.cy = res > 0xFFFF ? 1 : 0; setPair(2, res); } // MVI - Move Immediate else if ((currOp & 0xC7) === 0x06) { let dest = (currOp >> 3) & 0x07; setReg(dest, ram[cpu.pc++]); } // INR - Increment 8-Bit Register else if ((currOp & 0xC7) === 0x04) { let dest = (currOp >> 3) & 0x07; let val = (getReg(dest) + 1) & 0xFF; // INR updates AC for DAA. Simplified AC logic: cpu.ac = ((getReg(dest) & 0x0F) + 1 > 0x0F) ? 1 : 0; setReg(dest, val); updateFlags(val); } // DCR - Decrement 8-Bit Register else if ((currOp & 0xC7) === 0x05) { let dest = (currOp >> 3) & 0x07; let val = (getReg(dest) - 1) & 0xFF; setReg(dest, val); updateFlags(val); } // Arithmetic & Logical Operations (ADD, SUB, ANA, XRA, ORA, CMP) else if (currOp >= 0x80 && currOp <= 0xBF) { let src = getReg(currOp & 0x07); let type = (currOp >> 3) & 0x07; let res = cpu.a; if (type === 0) { // ADD res = cpu.a + src; cpu.ac = ((cpu.a & 0x0F) + (src & 0x0F) > 0x0F) ? 1 : 0; cpu.cy = res > 0xFF ? 1 : 0; cpu.a = res & 0xFF; updateFlags(cpu.a); } else if (type === 1) { // ADC res = cpu.a + src + cpu.cy; cpu.ac = ((cpu.a & 0x0F) + (src & 0x0F) + cpu.cy > 0x0F) ? 1 : 0; cpu.cy = res > 0xFF ? 1 : 0; cpu.a = res & 0xFF; updateFlags(cpu.a); } else if (type === 2) { // SUB res = cpu.a - src; cpu.cy = res < 0 ? 1 : 0; cpu.a = res & 0xFF; updateFlags(cpu.a); } else if (type === 3) { // SBB res = cpu.a - src - cpu.cy; cpu.cy = res < 0 ? 1 : 0; cpu.a = res & 0xFF; updateFlags(cpu.a); } else if (type === 4) { // ANA cpu.a &= src; cpu.cy = 0; updateFlags(cpu.a); } else if (type === 5) { // XRA cpu.a ^= src; cpu.cy = 0; updateFlags(cpu.a); } else if (type === 6) { // ORA cpu.a |= src; cpu.cy = 0; updateFlags(cpu.a); } else if (type === 7) { // CMP (Crucial for Sorting!) res = cpu.a - src; cpu.cy = res < 0 ? 1 : 0; updateFlags(res & 0xFF); // Does NOT alter Accumulator } } // Immediate Arithmetic & Logical (ADI, SUI, ANI, CPI, etc.) else if ([0xC6, 0xCE, 0xD6, 0xDE, 0xE6, 0xEE, 0xF6, 0xFE].includes(currOp)) { let src = ram[cpu.pc++]; let res = cpu.a; if (currOp === 0xC6) { // ADI res = cpu.a + src; cpu.ac = ((cpu.a & 0x0F) + (src & 0x0F) > 0x0F) ? 1 : 0; cpu.cy = res > 0xFF ? 1 : 0; cpu.a = res & 0xFF; updateFlags(cpu.a); } else if (currOp === 0xD6) { // SUI res = cpu.a - src; cpu.cy = res < 0 ? 1 : 0; cpu.a = res & 0xFF; updateFlags(cpu.a); } else if (currOp === 0xE6) { // ANI cpu.a &= src; cpu.cy = 0; updateFlags(cpu.a); } else if (currOp === 0xF6) { // ORI cpu.a |= src; cpu.cy = 0; updateFlags(cpu.a); } else if (currOp === 0xFE) { // CPI res = cpu.a - src; cpu.cy = res < 0 ? 1 : 0; updateFlags(res & 0xFF); } } // Special Accumulator / Machine Control else if (currOp === 0x2F) { // CMA (Crucial for 2's Complement) cpu.a = (~cpu.a) & 0xFF; } else if (currOp === 0x37) { // STC cpu.cy = 1; } else if (currOp === 0x3F) { // CMC cpu.cy = cpu.cy === 1 ? 0 : 1; } else if (currOp === 0x27) { // DAA (Critical for BCD Conversion Exp #15) let tempA = cpu.a; if ((tempA & 0x0F) > 9 || cpu.ac === 1) { tempA += 0x06; cpu.ac = ((cpu.a & 0x0F) + 6 > 0x0F) ? 1 : 0; } if ((tempA & 0xF0) > 0x90 || cpu.cy === 1) { tempA += 0x60; cpu.cy = 1; } cpu.a = tempA & 0xFF; updateFlags(cpu.a); } // Rotations else if (currOp === 0x07) { // RLC cpu.cy = cpu.a >> 7; cpu.a = ((cpu.a << 1) | cpu.cy) & 0xFF; } else if (currOp === 0x0F) { // RRC cpu.cy = cpu.a & 1; cpu.a = ((cpu.a >> 1) | (cpu.cy << 7)) & 0xFF; } else if (currOp === 0x17) { // RAL let oldCy = cpu.cy; cpu.cy = cpu.a >> 7; cpu.a = ((cpu.a << 1) | oldCy) & 0xFF; } else if (currOp === 0x1F) { // RAR let oldCy = cpu.cy; cpu.cy = cpu.a & 1; cpu.a = ((cpu.a >> 1) | (oldCy << 7)) & 0xFF; } // STA / LDA / SHLD / LHLD else if (currOp === 0x3A) { let l = ram[cpu.pc++]; let h = ram[cpu.pc++]; cpu.a = ram[(h << 8) | l]; } else if (currOp === 0x32) { let l = ram[cpu.pc++]; let h = ram[cpu.pc++]; ram[(h << 8) | l] = cpu.a; } else if (currOp === 0x2A) { let l = ram[cpu.pc++]; let h = ram[cpu.pc++]; let addr = (h << 8) | l; cpu.l = ram[addr]; cpu.h = ram[addr+1]; } else if (currOp === 0x22) { let l = ram[cpu.pc++]; let h = ram[cpu.pc++]; let addr = (h << 8) | l; ram[addr] = cpu.l; ram[addr+1] = cpu.h; } // Indirect Loading/Storing else if (currOp === 0x0A) { cpu.a = ram[(cpu.b << 8) | cpu.c]; } // LDAX B else if (currOp === 0x1A) { cpu.a = ram[(cpu.d << 8) | cpu.e]; } // LDAX D else if (currOp === 0x02) { ram[(cpu.b << 8) | cpu.c] = cpu.a; } // STAX B else if (currOp === 0x12) { ram[(cpu.d << 8) | cpu.e] = cpu.a; } // STAX D // Jumps else if (currOp === 0xC3) { let l = ram[cpu.pc++]; let h = ram[cpu.pc++]; cpu.pc = (h << 8) | l; } else if (currOp === 0xCA) { let l = ram[cpu.pc++]; let h = ram[cpu.pc++]; if (cpu.z === 1) cpu.pc = (h << 8) | l; } // JZ else if (currOp === 0xC2) { let l = ram[cpu.pc++]; let h = ram[cpu.pc++]; if (cpu.z === 0) cpu.pc = (h << 8) | l; } // JNZ else if (currOp === 0xDA) { let l = ram[cpu.pc++]; let h = ram[cpu.pc++]; if (cpu.cy === 1) cpu.pc = (h << 8) | l; } // JC else if (currOp === 0xD2) { let l = ram[cpu.pc++]; let h = ram[cpu.pc++]; if (cpu.cy === 0) cpu.pc = (h << 8) | l; } // JNC // Calls & Returns else if (currOp === 0xCD) { let l = ram[cpu.pc++]; let h = ram[cpu.pc++]; push((cpu.pc >> 8) & 0xFF, cpu.pc & 0xFF); cpu.pc = (h << 8) | l; } else if (currOp === 0xC9) { let p = pop(); cpu.pc = (p.h << 8) | p.l; } limit++; } }
